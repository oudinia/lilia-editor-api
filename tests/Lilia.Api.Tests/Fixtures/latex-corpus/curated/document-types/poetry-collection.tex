\begin{center}
  {\Large\bfseries Three Poems}\\[0.2cm]
  {\itshape by J. Avery}
\end{center}

\vspace{0.5cm}

\begin{center}
  \textbf{Morning}
\end{center}

\begin{verse}
Light arrives like a quiet guest, \\
stepping past curtains without a sound, \\
settling into chairs \\
before the house begins to stir.
\end{verse}

\vspace{0.5cm}

\begin{center}
  \textbf{River}
\end{center}

\begin{verse}
The river does not argue \\
with the stones it meets; \\
it turns, it murmurs, it continues. \\
We might learn from this.
\end{verse}

\vspace{0.5cm}

\begin{center}
  \textbf{Evening Walk}
\end{center}

\begin{verse}
The lamps have all been lit. \\
I walk past houses whose windows \\
hold small rooms of light, \\
each its own brief narrative.
\end{verse}
