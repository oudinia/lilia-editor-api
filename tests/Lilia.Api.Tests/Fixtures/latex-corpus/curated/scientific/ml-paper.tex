\begin{abstract}
We propose a low-rank approximation to the attention matrix that reduces complexity from $O(n^2)$ to $O(nr)$ where $r \ll n$, with minimal loss on downstream tasks.
\end{abstract}

\section{Introduction}
Transformer attention scales quadratically in sequence length, making long-context inference expensive.

\section{Method}

\begin{algorithm}
\caption{Low-rank attention}
\begin{algorithmic}[1]
\State Compute $Q, K, V = XW_Q, XW_K, XW_V$
\State Let $U, \Sigma, V^\top = \text{SVD}_r(K)$ \Comment{rank $r$ truncation}
\State Compute $\tilde K = U\Sigma$
\State Output $\text{softmax}(Q\tilde K^\top / \sqrt{d}) V$
\end{algorithmic}
\end{algorithm}

\section{Complexity}
Softmax over the rank-$r$ surrogate runs in $O(nrd)$ time. For $r = O(\log n)$ this is near-linear.
